% Guardião Soberano — ajustes LaTeX para A4 (margens + código + tabelas)
\usepackage{fvextra}
\usepackage{xurl}
\usepackage{microtype}
\usepackage{needspace}
\usepackage{ragged2e}

% Quebra de linha em blocos de código (shell, PGP) — tamanho legível
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{%
  breaklines,commandchars=\\\{\}}
\RecustomVerbatimEnvironment{verbatim}{Verbatim}{%
  breaklines}

% Tabelas: colunas proporcionais em A4 (largura útil ~175 mm)
\usepackage{tabularx}
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{array}
\usepackage{ltablex}
\keepXColumns
\newcolumntype{Y}{>{\raggedright\arraybackslash}X}
\renewcommand{\arraystretch}{1.15}
\setlength{\tabcolsep}{3pt}

% Evitar overflow horizontal de tabelas largas
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\small}
\AtBeginEnvironment{table}{\small}
\AtBeginEnvironment{table}{\setlength{\tabcolsep}{3pt}}

% Quebras de página: menos títulos órfãos e viúvas
\clubpenalty=10000
\widowpenalty=10000
\displaywidowpenalty=10000

% needspace antes de seções (## / ### no Markdown)
\pretocmd{\section}{\Needspace{5\baselineskip}}{}{}
\pretocmd{\subsection}{\Needspace{4\baselineskip}}{}{}
\pretocmd{\subsubsection}{\Needspace{3\baselineskip}}{}{}
\pretocmd{\paragraph}{\Needspace{2.5\baselineskip}}{}{}

% Blockquotes (AVISO, citações) — evitar bloco curto partido
\AtBeginEnvironment{quote}{\Needspace{4\baselineskip}}

% Tabelas: não iniciar no rodapé sem espaço para cabeçalho + 2 linhas
\BeforeBeginEnvironment{longtable}{\Needspace{3\baselineskip}}
\setlength{\LTpre}{0.6\baselineskip plus 0.15\baselineskip minus 0.1\baselineskip}
\setlength{\LTpost}{0.6\baselineskip plus 0.15\baselineskip minus 0.1\baselineskip}

% Pacote calc (colunas proporcionais geradas pelo Pandoc)
\usepackage{calc}

% URLs longas em tabelas e texto
\Urlmuskip=0mu plus 1mu

% Tipografia IBM Plex (recursos/fonts/) — fallback DejaVu se ausente
% Requer \GuardiaoFontPath definido em font-path.tex (gerado por compila.ps1)
\IfFileExists{\GuardiaoFontPath IBMPlexSerif-Regular.ttf}{%
  \setmainfont{IBMPlexSerif-Regular.ttf}[
    Path=\GuardiaoFontPath,
    BoldFont=IBMPlexSerif-Bold.ttf,
    ItalicFont=IBMPlexSerif-Italic.ttf,
    BoldItalicFont=IBMPlexSerif-BoldItalic.ttf,
    Ligatures=TeX
  ]
  \setsansfont{IBMPlexSans-Regular.ttf}[
    Path=\GuardiaoFontPath,
    BoldFont=IBMPlexSans-Bold.ttf,
    Ligatures=TeX
  ]
  \setmonofont{IBMPlexMono-Regular.ttf}[
    Path=\GuardiaoFontPath,
    BoldFont=IBMPlexMono-Bold.ttf,
    Scale=0.85,
    Ligatures=TeX
  ]
}{%
  \setmainfont{DejaVu Serif}
  \setsansfont{DejaVu Sans}
  \setmonofont{DejaVu Sans Mono}[Scale=0.85]
}

% Títulos em sans (IBM Plex Sans), corpo em serif
\usepackage{sectsty}
\allsectionsfont{\sffamily\bfseries}
